def wait_for_ajax
  Timeout.timeout(Capybara.default_wait_time) do
    active = page.evaluate_script('$("#flash").is(":visible")')
    until active == true
      active = page.evaluate_script('$("#flash").is(":visible")')
    end
  end
end
