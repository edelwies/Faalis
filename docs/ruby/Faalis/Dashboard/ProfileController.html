def update_password
  @user = ::Faalis::User.find(current_user.id)
  authorize @user
  respond_to do |f|
    if @user.update_with_password(resource_params)
      f.js { redirect_to new_user_session_path }
      f.html { redirect_to new_user_session_path }
    else
      @errors = @user.errors
      f.js {render :errors}
      f.html
    end
  end
end
