def sidebar(name, **options, &block)
  side = Faalis::Dashboard::Models::Sidebar.new(name, **options)

  side.instance_eval do
    block.call if block_given?
  end

  define_method(:setup_sidebar) do
    instance_variable_set('@sidebar', side)
  end
end
