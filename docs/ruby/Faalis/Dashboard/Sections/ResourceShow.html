def show
  @resource = model.find(params[:id])
  authorize @resource

  collect_model_fields_for_show

  @resource_title = _resource_title.singularize

  show_hook(@resource)

  return if _override_views.include? :show
  render 'faalis/dashboard/resource/show'
end
