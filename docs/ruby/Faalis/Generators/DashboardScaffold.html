def source_paths
  @source_paths = self.class.source_paths_for_search

  paths = []
  Faalis::Extension.extensions.each do |name, klass|
    if klass.respond_to? :generator_templates_path.to_sym
      paths << klass.generator_templates_path
    end
  end
  paths += @source_paths
  @source_paths = paths
end
