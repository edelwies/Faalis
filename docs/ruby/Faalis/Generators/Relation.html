def restangular
  result = "API"
  if options.include? "parents"
    field_parents.each do |parent|
      result = "#{result}.one('#{parent}', #{}_id)"
    end
  end
  to.split("/").each do |resource|
    result = "#{result}.all(\"#{resource}\")"
  end
  result
end
