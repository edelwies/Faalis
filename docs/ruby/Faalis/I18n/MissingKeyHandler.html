def create_key_cache(locale, key)
  locale_dir = "#{Rails.root}/tmp/i18n/#{locale}"
  key_file   = "#{locale_dir}/#{key}"

  FileUtils.mkdir_p locale_dir
  FileUtils.touch [key_file]
end
