def set_locale
  if I18n.available_locales.length > 1
    lang = request.env['lang'] || params[:locale] || session[:locale] || I18n.default_locale
  else
    lang = I18n.default_locale
  end

  session[:locale] = I18n.locale = :#{lang}"
end
